        function openModal() {
            const modal = document.getElementById('triggerModal');
            modal.style.display = 'block';
            refreshTriggerList();
        }

        function closeModal() {
            const modal = document.getElementById('triggerModal');
            modal.style.display = 'none';
        }

        function refreshTriggerList() {
            const listContainer = document.getElementById('triggerList');
            listContainer.innerHTML = '';
            
            for (let key in messagingData) {
                const item = document.createElement('div');
                item.className = 'trigger-list-item';
                item.innerHTML = `
                    <span class="trigger-name" contenteditable="true" onblur="updateTriggerName('${key}', this.textContent)">${messagingData[key].name}</span>
                    <button class="delete-trigger-btn" onclick="deleteTrigger('${key}')">üóëÔ∏è Delete</button>
                `;
                listContainer.appendChild(item);
            }
        }

        function updateTriggerName(key, newName) {
            messagingData[key].name = newName.trim();
            populateDropdown();
        }

        function deleteTrigger(key) {
            if (Object.keys(messagingData).length <= 1) {
                alert('You must keep at least one trigger event.');
                return;
            }
            if (confirm('Are you sure you want to delete this trigger event?')) {
                delete messagingData[key];
                populateDropdown();
                refreshTriggerList();
                loadTrigger();
            }
        }

        function openTab(evt, tabName) {
            var tabContents = document.getElementsByClassName("tab-content");
            for (var i = 0; i < tabContents.length; i++) {
                tabContents[i].classList.remove("active");
            }
            var tabs = document.getElementsByClassName("tab");
            for (var i = 0; i < tabs.length; i++) {
                tabs[i].classList.remove("active");
            }
            document.getElementById(tabName).classList.add("active");
            evt.currentTarget.classList.add("active");
            
            if (tabName === 'messaging') {
                loadTrigger();
            }
        }
